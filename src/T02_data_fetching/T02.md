
# Zustand

- `npm install zustand`

---

`./components/utils/store.ts`

```ts
import { create } from "zustand";
import { type Time } from "./types";

interface Store {
  time: Time | null;
  setTime: (time: Time) => void;
}

const useStore = create<Store>((set) => ({
  time: null,
  setTime: (time) => set({ time }),
}));

export default useStore;
```
